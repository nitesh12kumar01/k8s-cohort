auditor-role.yaml:

apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  namespace: development
  name: pod-lister
rules:
- apiGroups: [""] # "" indicates the core API group
  resources: ["pods"]
  verbs: ["get", "list", "watch"]

-----
  kubectl apply -f auditor-role.yaml
-----
  kubectl create rolebinding auditor-binding \
  --role=pod-lister \
  --serviceaccount=development:pod-auditor \
  -n development

-----
The RoleBinding is the "glue" that grants the Role to the ServiceAccount.

kubectl create rolebinding auditor-binding \
  --role=pod-lister \
  --serviceaccount=development:pod-auditor \
  -n development
---

1. Test "Positive" Permission
Ask Kubernetes: "Can the auditor list pods?"

kubectl auth can-i list pods \
  --as=system:serviceaccount:development:pod-auditor \
  -n development
Expected Output: yes

2. Test "Negative" Permission
Ask Kubernetes: "Can the auditor see secrets?"

kubectl auth can-i list secrets \
  --as=system:serviceaccount:development:pod-auditor \
  -n development
Expected Output: no

3. Test Namespace Isolation
RBAC Roles are namespaced. Ask: "Can this auditor see pods in the production namespace?"


kubectl auth can-i list pods \
  --as=system:serviceaccount:development:pod-auditor \
  -n production
Expected Output: no (Even though they are a "pod-lister," they are only bound in development).

clusterrole and clusterrolebinding
----

apiVersion: v1
kind: ServiceAccount
metadata:
  name: node-reader-sa
  namespace: default


apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
  name: node-reader-role
rules:
- apiGroups: [""] # "" indicates the core API group
  resources: ["nodes"]
  verbs: ["get", "watch", "list"]


apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRoleBinding
metadata:
  name: read-nodes-globally
subjects:
- kind: ServiceAccount
  name: node-reader-sa
  namespace: default # The namespace where the SA lives
roleRef:
  kind: ClusterRole
  name: node-reader-role
  apiGroup: rbac.authorization.k8s.io

----
Check if the default ServiceAccount can see nodes:

kubectl auth can-i list nodes --as=system:serviceaccount:default:default
# Expected output: no

Check if your new ServiceAccount can see nodes:

kubectl auth can-i list nodes --as=system:serviceaccount:default:node-reader-sa
# Expected output: yes
